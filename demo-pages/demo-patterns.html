Event Logger

Interact with any component below to see Scooter events in the console. Demonstrates the sc:* event system.

// Interact with the controls above…
<!-- Listen for all sc:* events -->
<script>
  const log = document.getElementById('event-log');
  ['sc:change','sc:toggle','sc:select','sc:slide',
   'sc:open','sc:close','sc:action','sc:complete'
  ].forEach(type => {
    document.addEventListener(type, e => {
      const line = document.createElement('div');
      line.textContent = \`[\${type}] \${JSON.stringify(e.detail)}\`;
      log.appendChild(line);
      log.scrollTop = log.scrollHeight;
    });
  });
</script>
