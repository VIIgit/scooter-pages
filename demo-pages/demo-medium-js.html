Dialog

Modal dialog with overlay. Closes on backdrop click, Escape key, and the built-in close button. Traps focus while open.

Edit Profile
Make changes to your profile here. Click save when you're done.

Opens a centered modal with overlay — try Escape or click backdrop to close

// dialog events appear here…
<div data-sc="dialog">
  <button data-slot="dialog-trigger"
          data-slot-also="button" data-variant="outline">
    Edit Profile
  </button>
  <dialog data-slot="dialog-content">
    <div data-slot="dialog-header">
      <div data-slot="dialog-title">Edit Profile</div>
      <div data-slot="dialog-description">
        Make changes here. Click save when done.
      </div>
    </div>
    <div style="padding:16px 0">
      <label data-slot="label">Name</label>
      <input data-slot="input" value="Pedro Duarte">
    </div>
    <div data-slot="dialog-footer">
      <button data-slot="button" id="dialog-save">Save</button>
    </div>
  </dialog>
</div>
// Get reference
const dialog = document.querySelector('[data-sc="dialog"]');

// Programmatic open / close
dialog._dialog.open();
dialog._dialog.close();

// Listen for open & close events
dialog.addEventListener('sc:open', () => {
  console.log('Dialog opened');
});
dialog.addEventListener('sc:close', () => {
  console.log('Dialog closed');
});

// Handle the Save button inside the dialog
document.getElementById('dialog-save')
  .addEventListener('click', () => {
    const name = document.getElementById('dialog-name').value;
    console.log('Saving name:', name);
    dialog._dialog.close(); // close after save
  });

Alert Dialog

Confirmation dialog. Cannot be dismissed via backdrop click or Escape — the user must click Cancel or Confirm.

Are you absolutely sure?
This action cannot be undone. This will permanently delete your account and remove your data from our servers.

No Escape / backdrop close — forces an explicit Cancel or Confirm action

// alert-dialog events appear here…
<div data-sc="alert-dialog">
  <button data-slot="alert-dialog-trigger"
          data-slot-also="button" data-variant="destructive">
    Delete Account
  </button>
  <dialog data-slot="alert-dialog-content">
    <div data-slot="alert-dialog-header">
      <div data-slot="alert-dialog-title">Are you sure?</div>
      <div data-slot="alert-dialog-description">
        This action cannot be undone.
      </div>
    </div>
    <div data-slot="alert-dialog-footer">
      <button data-slot="alert-dialog-cancel"
              data-slot-also="button" data-variant="outline">
        Cancel
      </button>
      <button data-slot="alert-dialog-action"
              data-slot-also="button" data-variant="destructive">
        Yes, delete
      </button>
    </div>
  </dialog>
</div>
const alertDialog = document.querySelector('[data-sc="alert-dialog"]');

// The "action" button fires sc:action
alertDialog.addEventListener('sc:action', () => {
  console.log('User confirmed! Performing destructive action…');
  // e.g. fetch('/api/account', { method: 'DELETE' });
});

// The "cancel" button fires sc:cancel
alertDialog.addEventListener('sc:cancel', () => {
  console.log('User cancelled — no action taken.');
});

// Programmatic open
alertDialog._alertDialog.open();

const menu = document.querySelector('[data-sc="dropdown-menu"]');

// Listen for item selection
menu.addEventListener('sc:select', (e) => {
  console.log('Selected:', e.detail.value);

  switch (e.detail.value) {
    case 'profile':
      window.location.href = '/profile';
      break;
    case 'settings':
      window.location.href = '/settings';
      break;
    case 'logout':
      fetch('/api/logout', { method: 'POST' })
        .then(() => window.location.href = '/login');
      break;
    case 'invite-email':
      openInviteDialog('email');
      break;
  }
});

const select = document.querySelector('[data-sc="select"]');

// Listen for value changes
select.addEventListener('sc:change', (e) => {
  console.log('Selected:', e.detail.value);
  // e.detail.value → "apple", "banana", etc.
});

// Get current value programmatically
const currentValue = select._select.getValue();
console.log('Current value:', currentValue);

// Set value programmatically
select._select.setValue('cherry');

// Use in forms: read the value on submit
document.querySelector('form').addEventListener('submit', (e) => {
  e.preventDefault();
  const food = select._select.getValue();
  console.log('Form submitted with:', food);
});

  const link = e.target.closest('[data-slot="navigation-menu-link"]');
  if (link) {
    e.preventDefault();
    const route = link.getAttribute('data-value') || link.href;

// Example: highlight active link
function setActiveLink(value) {
  nav.querySelectorAll('[data-slot="navigation-menu-link"]')
    .forEach(link => {
      link.classList.toggle('active',
        link.getAttribute('data-value') === value);
    });
}
