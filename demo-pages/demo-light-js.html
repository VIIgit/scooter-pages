Accordion

Expandable sections. Supports data-type="single|multiple" and data-collapsible.

Yes. It adheres to the WAI-ARIA design pattern for accordions.
Yes. It ships with default styles from base.css that match the other components.
Yes. It uses CSS height transitions with measured content height for smooth open/close. 
<div data-sc="accordion" data-type="single" data-collapsible>
  <div data-slot="accordion-item" data-value="item-1">
    <button data-slot="accordion-trigger" aria-expanded="false">
      Question text
      <svg …><!-- chevronDown --></svg>
    </button>
    <div data-slot="accordion-content" data-state="closed">
      <div data-slot="accordion-content-inner">
        Answer text
      </div>
    </div>
  </div>
</div>

<!-- data-type="multiple" allows multiple open at once -->
const accordion = document.querySelector('[data-sc="accordion"]');

// Listen for item open/close
accordion.addEventListener('sc:change', (e) => {
  console.log('Active item:', e.detail.value);
  // e.detail.value → "item-1", "item-2", …
});

// Programmatic control
accordion._accordion.open('item-2');  // expand a specific item
accordion._accordion.closeAll();      // collapse everything

  console.log('Checked:', e.detail.checked);
  // Example: enable/disable a submit button
  submitBtn.disabled = !e.detail.checked;
});

// Read state programmatically
const isChecked = checkbox.getAttribute('aria-checked') === 'true';

// Example: "select all" pattern

  .addEventListener('sc:change', (e) => {
    document.querySelectorAll('.item-checkbox')
      .forEach(cb => cb._checkbox.setChecked(e.detail.checked));
  });

// Button Group — single select behavior
const group = document.querySelector('[data-slot="button-group"]');
group.addEventListener('click', (e) => {
  const btn = e.target.closest('[data-slot="button"]');
  if (!btn) return;
  // Deselect all, select clicked
  group.querySelectorAll('[data-slot="button"]').forEach(b => {
    b.setAttribute('aria-pressed', 'false');
  });
  btn.setAttribute('aria-pressed', 'true');
  console.log('Selected:', btn.dataset.value);
});

// Toggle loading state programmatically
function setLoading(btn, loading) {
  if (loading) {
    btn.setAttribute('data-loading', '');
    btn.innerHTML = '<span class="button-spinner"></span> Saving…';
  } else {
    btn.removeAttribute('data-loading');
    btn.innerHTML = 'Save';
  }
}
