Menubar

Desktop-style menu bar with multiple top-level menus. Arrow key navigation between triggers.

<div data-sc="menubar" data-slot="menubar">
  <button data-slot="menubar-trigger">File</button>
  <template data-slot="menubar-content" data-for="File">
    <div data-slot="menubar-item">New Tab
      <span data-slot="menubar-shortcut">⌘T</span>
    </div>
    <div data-slot="menubar-separator"></div>
    <div data-slot="menubar-item">Print…</div>
  </template>

  <button data-slot="menubar-trigger">Edit</button>
  <template data-slot="menubar-content" data-for="Edit">…</template>
</div>
// Get reference
const menubar = document.querySelector('[data-sc="menubar"]');

// Listen for item selection
menubar.addEventListener('sc:select', e => {
  console.log('Selected:', e.detail);
  // e.detail = text content of the clicked item
});

// Programmatic API
menubar._menubar.showMenu(0);   // open first menu (File)
menubar._menubar.hideAll();     // close any open menu

      <div data-slot="carousel-item">Slide 2</div>
      <div data-slot="carousel-item">Slide 3</div>
    </div>

const carousel = document.querySelector('[data-sc="carousel"]');

// Listen for slide changes
carousel.addEventListener('sc:slide', e => {
  console.log('Current slide:', e.detail.current);
});

// Programmatic API
carousel._carousel.next();          // advance to next slide
carousel._carousel.prev();          // go to previous slide
carousel._carousel.goTo(2);         // jump to slide at index 2
carousel._carousel.getCurrent();    // returns current index

// Attributes: data-loop, data-auto-play="3000", data-orientation

Calendar

Self-rendering date picker. Just an empty container — JS builds the entire UI. Emits sc:change.

<!-- Self-rendering: just provide empty div -->
<div data-sc="calendar"></div>
const calendar = document.querySelector('[data-sc="calendar"]');

// Listen for date selection
calendar.addEventListener('sc:change', e => {
  console.log('Selected date:', e.detail);
  // e.detail = "YYYY-MM-DD"
});

// Programmatic API
calendar._calendar.getValue();            // "2026-03-16" or ""
calendar._calendar.setValue('2026-12-25'); // set date & re-render
calendar._calendar.goTo(2026, 5);         // navigate to June 2026 (0-based month)

Form

Declarative validation via HTML5 attributes. Emits sc:action with cleaned data on submit. Shows inline errors.

This is your public display name.

Must match the password field.

<form data-sc="form">
  <div data-slot="form-item">
    <label data-slot="form-label">Username</label>
    <div data-slot="form-control">
      <input data-slot="input" name="username" required minlength="3">
    </div>
    <p data-slot="form-description">Help text.</p>
    <p data-slot="form-message"></p>
  </div>

  <!-- data-match="password" to cross-validate -->
  <input name="confirm" data-match="password">

  <button type="submit">Submit</button>
</form>
const form = document.querySelector('[data-sc="form"]');

// Listen for valid form submission
form.addEventListener('sc:action', e => {
  const data = Object.fromEntries(e.detail.data);
  console.log('Form data:', data);
  // { username: "john", email: "john@example.com", … }
});

// Programmatic API
form._form.validate();   // run all validations, returns true/false
form._form.reset();      // clear form & all error states

// Supports data-match="fieldName" for cross-field validation

Input OTP

Self-rendering OTP/PIN input. Set data-length and optional data-groups. Emits sc:complete.

6 digits, grouped 3-3

4 digits, single group

<!-- Self-rendering: just provide empty div -->
<div data-sc="input-otp" data-length="6" data-groups="3,3"></div>

<!-- 4-digit single group -->
<div data-sc="input-otp" data-length="4"></div>
const otp = document.querySelector('[data-sc="input-otp"]');

// Listen for completion (all slots filled)
otp.addEventListener('sc:complete', e => {
  console.log('OTP entered:', e.detail);
  // e.detail = "123456"
});

// Programmatic API
otp._inputOtp.getValue();          // current value as string
otp._inputOtp.setValue('123456');   // set value across slots
otp._inputOtp.clear();             // clear all slots & focus first
otp._inputOtp.focus();             // focus first slot

Toast

Imperative-only notifications. Call Scooter.toast(title, options). No HTML needed.

<!-- Toast is imperative — no HTML markup needed.
     Trigger via onclick or call from JavaScript. -->

<button data-slot="button" data-variant="outline"
  onclick="Scooter.toast('Hello!', { description: 'World' })">
  Show Toast
</button>
// Default toast
Scooter.toast('Event Created', {
  description: 'Sunday, December 03, 2023 at 9:00 AM'
});

// Success variant
Scooter.toast('Saved!', {
  description: 'Your changes have been saved.',
  variant: 'success'
});

// Destructive variant
Scooter.toast('Error', {
  description: 'Something went wrong.',
  variant: 'destructive'
});

// Custom duration (ms, default: 4000)
Scooter.toast('Slow', { duration: 10000 });

// Programmatic dismiss
const t = Scooter.toast('Processing…', { duration: 0 });
t.dismiss();   // dismiss manually

// Configure defaults
Scooter.toast.defaults.position = 'top-right';
Scooter.toast.defaults.duration = 6000;
Scooter.toast.defaults.maxToasts = 3;

const sem = document.querySelector('[data-sc="system-environment-matrix"]');

// Element click → selects element, filters to compact view of that env
sem.addEventListener('sc:element', e => {
  console.log(e.detail);
  // { environment: "Production", element: "Firewall", characteristic: "Security" }
});

// After filtering, click the env name to expand back to the grid
sem.addEventListener('sc:filter', e => {
  console.log('Filtered to', e.detail.environment);
});

// Click the background to reset to the initial grid view
sem.addEventListener('sc:reset', () => {
  console.log('Reset — back to full grid view');
});

// Programmatic API
sem._sem.filter('Production');   // filter to one environment
sem._sem.reset();                // clear filter, return to initial view
sem._sem.expand();               // force expanded grid
sem._sem.collapse();             // force collapsed tags
sem._sem.toggle();               // switch mode
sem._sem.selected;               // current env filter (string | null)
sem._sem.selectedElement;        // { environment, name, characteristic } | null
